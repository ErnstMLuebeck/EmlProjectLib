302 //Matrix Inversion Routine
303 // * This function inverts a matrix based on the Gauss Jordan method.
304 // * Specifically, it uses partial pivoting to improve numeric stability.
305 // * The algorithm is drawn from those presented in
306 // NUMERICAL RECIPES: The Art of Scientific Computing.
307 // * The function returns 1 on success, 0 on failure.
308 // * NOTE: The argument is ALSO the result matrix, meaning the input matrix is REPLACED
309 int StateSpaceModel::MatrixInvert(float* A, int n)
310 {
311  // A = input matrix AND result matrix
312  // n = number of rows = number of columns in A (n x n)
313  int pivrow = 0; // keeps track of current pivot row
314  int k, i, j; // k: overall index along diagonal; i: row index; j: col index
315  int pivrows[n]; // keeps track of rows swaps to undo at end
316  float tmp; // used for finding max value and making column swaps
317 
318  for (k = 0; k < n; k++)
319  {
320  // find pivot row, the row with biggest entry in current column
321  tmp = 0;
322  for (i = k; i < n; i++)
323  {
324  if (abs(A[i * n + k]) >= tmp) // 'Avoid using other functions inside abs()?'
325  {
326  tmp = abs(A[i * n + k]);
327  pivrow = i;
328  }
329  }
330 
331  // check for singular matrix
332  if (A[pivrow * n + k] == 0.0f)
333  {
334  Serial.println("Inversion failed due to singular matrix");
335  return 0;
336  }
337 
338  // Execute pivot (row swap) if needed
339  if (pivrow != k)
340  {
341  // swap row k with pivrow
342  for (j = 0; j < n; j++)
343  {
344  tmp = A[k * n + j];
345  A[k * n + j] = A[pivrow * n + j];
346  A[pivrow * n + j] = tmp;
347  }
348  }
349  pivrows[k] = pivrow; // record row swap (even if no swap happened)
350 
351  tmp = 1.0f / A[k * n + k]; // invert pivot element
352  A[k * n + k] = 1.0f; // This element of input matrix becomes result matrix
353 
354  // Perform row reduction (divide every element by pivot)
355  for (j = 0; j < n; j++)
356  {
357  A[k * n + j] = A[k * n + j] * tmp;
358  }
359 
360  // Now eliminate all other entries in this column
361  for (i = 0; i < n; i++)
362  {
363  if (i != k)
364  {
365  tmp = A[i * n + k];
366  A[i * n + k] = 0.0f; // The other place where in matrix becomes result mat
367  for (j = 0; j < n; j++)
368  {
369  A[i * n + j] = A[i * n + j] - A[k * n + j] * tmp;
370  }
371  }
372  }
373  }
374 
375  // Done, now need to undo pivot row swaps by doing column swaps in reverse order
376  for (k = n - 1; k >= 0; k--)
377  {
378  if (pivrows[k] != k)
379  {
380  for (i = 0; i < n; i++)
381  {
382  tmp = A[i * n + k];
383  A[i * n + k] = A[i * n + pivrows[k]];
384  A[i * n + pivrows[k]] = tmp;
385  }
386  }
387  }
388  return 1;
389 }
