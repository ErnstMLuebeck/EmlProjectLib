66 void calcSunAngle(float lati, float longi, int month, int day, int hour, int minute, float* azimut, float* elevation)
67 {
68  int days = date2day(day,month);
69 
70  float pi = 3.1415926536;
71  float K = pi/180.0;
72 
73  float deklination = -23.45*cos(K*360*(days+10)/365);
74  float zeitgleichung = 60*(-0.171*sin(0.0337*days + 0.465) - 0.1299 * sin(0.01787*days - 0.168));
75  float stundenwinkel = 15.0*(hour + minute/60.0 - (15.0-longi)/15.0 - 12.0 + zeitgleichung/60.0);
76  float sin_hoehe = sin(K*lati)*sin(K*deklination) + cos(K*lati)*cos(K*deklination)*cos(K*stundenwinkel);
77  float y = -(sin(K*lati) * sin_hoehe - sin(K*deklination)) / (cos(K*lati) * sin( acos(sin_hoehe)));
78 
79  *azimut = acos(y)/K;
80  *elevation = asin(sin_hoehe)/K;
81 
82  if(hour>=12) *azimut = 360 - *azimut;
83 
84  /* calculation validated agains MATLAB. Identical results. */
85 }
