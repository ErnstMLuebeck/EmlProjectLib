414  }
415

417  if (pivrow != k)
418  {

420  for (j = 0; j < n; j++)

424  A[pivrow * n + j] = tmp;

428

430  A[k * n + k] = 1.0f; // This element of input matrix becomes result matrix
431 
432  // Perform row reduction (divide every element by pivot)
433  for (j = 0; j < n; j++)
434  {
435  A[k * n + j] = A[k * n + j] * tmp;
436  }
437
