300 {
301  // A = input matrix AND result matrix
302  // n = number of rows = number of columns in A (n x n)
303  int pivrow = 0; // keeps track of current pivot row
304  int k, i, j; // k: overall index along diagonal; i: row index; j: col index
305  int pivrows[n]; // keeps track of rows swaps to undo at end
306  float tmp; // used for finding max value and making column swaps
307 
308  for (k = 0; k < n; k++)
309  {
310  // find pivot row, the row with biggest entry in current column
311  tmp = 0;
312  for (i = k; i < n; i++)
313  {
314  if (abs(A[i * n + k]) >= tmp) // 'Avoid using other functions inside abs()?'
315  {
316  tmp = abs(A[i * n + k]);
317  pivrow = i;
318  }
319  }
320 
321  // check for singular matrix
322  if (A[pivrow * n + k] == 0.0f)
323  {
324  Serial.println("Inversion failed due to singular matrix");
325  return 0;
326  }
327 
328  // Execute pivot (row swap) if needed
329  if (pivrow != k)
330  {
331  // swap row k with pivrow
332  for (j = 0; j < n; j++)
333  {
334  tmp = A[k * n + j];
335  A[k * n + j] = A[pivrow * n + j];
336  A[pivrow * n + j] = tmp;
337  }
338  }
339  pivrows[k] = pivrow; // record row swap (even if no swap happened)
340 
341  tmp = 1.0f / A[k * n + k]; // invert pivot element
342  A[k * n + k] = 1.0f; // This element of input matrix becomes result matrix
343 
344  // Perform row reduction (divide every element by pivot)
345  for (j = 0; j < n; j++)
346  {
347  A[k * n + j] = A[k * n + j] * tmp;
348  }
349 
350  // Now eliminate all other entries in this column
351  for (i = 0; i < n; i++)
352  {
353  if (i != k)
354  {
355  tmp = A[i * n + k];
356  A[i * n + k] = 0.0f; // The other place where in matrix becomes result mat
357  for (j = 0; j < n; j++)
358  {
359  A[i * n + j] = A[i * n + j] - A[k * n + j] * tmp;
360  }
361  }
362  }
363  }
364 
365  // Done, now need to undo pivot row swaps by doing column swaps in reverse order
366  for (k = n - 1; k >= 0; k--)
367  {
368  if (pivrows[k] != k)
369  {
370  for (i = 0; i < n; i++)
371  {

374  A[i * n + pivrows[k]] = tmp;
375  }
376  }
