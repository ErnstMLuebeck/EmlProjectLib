342 {

344  // n = number of rows = number of columns in A (n x n)
345  int pivrow = 0; // keeps track of current pivot row
346  int k, i, j; // k: overall index along diagonal; i: row index; j: col index
347  int pivrows[n]; // keeps track of rows swaps to undo at end
348  float tmp; // used for finding max value and making column swaps
349 
350  for (k = 0; k < n; k++)
351  {
352  // find pivot row, the row with biggest entry in current column
353  tmp = 0;
354  for (i = k; i < n; i++)
355  {
356  if (abs(A[i * n + k]) >= tmp) // 'Avoid using other functions inside abs()?'
357  {

360  }
361  }

363  // check for singular matrix
364  if (A[pivrow * n + k] == 0.0f)
365  {
366  Serial.println("Inversion failed due to singular matrix");
367  return 0;
368  }
369 
370  // Execute pivot (row swap) if needed
371  if (pivrow != k)
372  {
373  // swap row k with pivrow
374  for (j = 0; j < n; j++)
375  {
376  tmp = A[k * n + j];
377  A[k * n + j] = A[pivrow * n + j];
378  A[pivrow * n + j] = tmp;
379  }
380  }
381  pivrows[k] = pivrow; // record row swap (even if no swap happened)
382 
383  tmp = 1.0f / A[k * n + k]; // invert pivot element
384  A[k * n + k] = 1.0f; // This element of input matrix becomes result matrix
385 
386  // Perform row reduction (divide every element by pivot)
387  for (j = 0; j < n; j++)
388  {
389  A[k * n + j] = A[k * n + j] * tmp;
390  }
391 
392  // Now eliminate all other entries in this column
393  for (i = 0; i < n; i++)
394  {
395  if (i != k)
396  {
397  tmp = A[i * n + k];
398  A[i * n + k] = 0.0f; // The other place where in matrix becomes result mat
399  for (j = 0; j < n; j++)

402  }
403  }
404  }
405  }

408  for (k = n - 1; k >= 0; k--)

411  {
412  for (i = 0; i < n; i++)

416  A[i * n + pivrows[k]] = tmp;
